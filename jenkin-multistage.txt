pipeline {
    agent any

    stages {
        stage('git checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/subhansu7/Terraform-CI-CD.git'
            }
        }
        stage('terraform init') {
            steps {
                sh 'terraform init'
            }
        }
        stage('terraform plan') {
            steps {
              sh 'terraform plan'
            }
        }
        stage('terraform apply') {
            steps {
                sh 'terraform destroy -auto-approve'
            }
        }
    }
}
